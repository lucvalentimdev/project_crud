CREATE TABLE cliente (
    id_cliente      INTEGER    PRIMARY KEY AUTOINCREMENT
                               NOT NULL,
    nome            TEXT (100) NOT NULL,
    telefone        TEXT (30),
    data_nascimento TEXT (10),
    tipo_cliente    INTEGER,
    cpf             TEXT (11),
    cnpj            TEXT (14),
    rg              TEXT (11),
    email           TEXT (100),
    endereco        TEXT (150),
    cidade          TEXT (100),
    CEP             TEXT (8) 
);
