    id_pessoa      INTEGER PRIMARY KEY AUTOINCREMENT,
    nome           TEXT (100) NOT NULL,
    telefone       TEXT (30),
    data_nascimento TEXT, -- Usar formato YYYY-MM-DD
    tipo_cliente   INTEGER REFERENCES TipoPessoa(id),
    cpf            TEXT (11),
    cnpj           TEXT (14),
    rg             TEXT (11),
    email          TEXT (100),
    endereco       TEXT (150),
    cidade         TEXT (100),
    estado	   TEXT (100),
    CEP            TEXT (9), -- Com m√°scara
    end_numero      TEXT
);
